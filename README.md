## Usage

```js
const { OIDCWebClient } = require('oidc-web')

let auth = new OIDCWebClient({})

auth.login(issuer)
  .then(session => {
    // logged in session or null
  })

// On Document Ready (or on web framework ready) event:

auth.currentSession()
  .then(session => {
    if (session) {
      // logged in
    } else {
      console.log('please log in')
    }
  })
```
